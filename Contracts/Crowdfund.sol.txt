pragma solidity ^0.8.19;

contract Crowdfund {
    struct Campaign {
        address creator;
        string title;
        string description;
        uint256 goal;
        uint256 raised;
        uint256 deadline;
        bool claimed;
        bool cancelled;
    }

    uint256 public campaignCount;
    mapping(uint256 => Campaign) public campaigns;
    mapping(uint256 => mapping(address => uint256)) public contributions;
    mapping(address => uint256[]) public creatorCampaigns;
    mapping(address => uint256[]) public contributorCampaigns;

    event CampaignCreated(uint256 indexed id, address indexed creator, string title, uint256 goal, uint256 deadline);
    event Contributed(uint256 indexed id, address indexed contributor, uint256 amount, uint256 totalRaised);
    event GoalReached(uint256 indexed id, uint256 totalRaised);
    event FundsClaimed(uint256 indexed id, address indexed creator, uint256 amount);
    event Refunded(uint256 indexed id, address indexed contributor, uint256 amount);
    event CampaignCancelled(uint256 indexed id);

    // Anyone can create a campaign
    function createCampaign(
        string calldata title,
        string calldata description,
        uint256 goalInWei,
        uint256 durationInDays
    ) external returns (uint256) {
        require(bytes(title).length > 0, "Title required");
        require(bytes(title).length <= 100, "Title too long");
        require(goalInWei > 0, "Goal must be > 0");
        require(durationInDays >= 1 && durationInDays <= 90, "Duration must be 1-90 days");

        campaignCount++;

        campaigns[campaignCount] = Campaign({
            creator: msg.sender,
            title: title,
            description: description,
            goal: goalInWei,
            raised: 0,
            deadline: block.timestamp + (durationInDays * 1 days),
            claimed: false,
            cancelled: false
        });

        creatorCampaigns[msg.sender].push(campaignCount);

        emit CampaignCreated(
            campaignCount,
            msg.sender,
            title,
            goalInWei,
            campaigns[campaignCount].deadline
        );

        return campaignCount;
    }

    // Anyone can contribute ETH to any active campaign
    function contribute(uint256 campaignId) external payable {
        Campaign storage c = campaigns[campaignId];
        require(c.creator != address(0), "Campaign does not exist");
        require(!c.cancelled, "Campaign was cancelled");
        require(!c.claimed, "Campaign already completed");
        require(block.timestamp < c.deadline, "Campaign has ended");
        require(msg.value > 0, "Must send ETH");

        c.raised += msg.value;
        contributions[campaignId][msg.sender] += msg.value;
        contributorCampaigns[msg.sender].push(campaignId);

        emit Contributed(campaignId, msg.sender, msg.value, c.raised);

        if (c.raised >= c.goal) {
            emit GoalReached(campaignId, c.raised);
        }
    }

    // Creator claims funds after deadline if goal reached
    function claimFunds(uint256 campaignId) external {
        Campaign storage c = campaigns[campaignId];
        require(c.creator != address(0), "Campaign does not exist");
        require(msg.sender == c.creator, "Only creator can claim");
        require(block.timestamp >= c.deadline, "Campaign still active");
        require(c.raised >= c.goal, "Goal not reached");
        require(!c.claimed, "Already claimed");
        require(!c.cancelled, "Campaign cancelled");

        c.claimed = true;
        uint256 amount = c.raised;

        (bool sent,) = payable(c.creator).call{value: amount}("");
        require(sent, "Transfer failed");

        emit FundsClaimed(campaignId, c.creator, amount);
    }

    // Anyone gets refund if goal not reached or campaign cancelled
    function refund(uint256 campaignId) external {
        Campaign storage c = campaigns[campaignId];
        require(c.creator != address(0), "Campaign does not exist");
        require(
            c.cancelled || block.timestamp >= c.deadline,
            "Campaign still active"
        );
        require(
            c.cancelled || c.raised < c.goal,
            "Goal was reached, no refund"
        );

        uint256 amount = contributions[campaignId][msg.sender];
        require(amount > 0, "No contribution to refund");

        contributions[campaignId][msg.sender] = 0;

        (bool sent,) = payable(msg.sender).call{value: amount}("");
        require(sent, "Refund failed");

        emit Refunded(campaignId, msg.sender, amount);
    }

    // Only creator can cancel before deadline
    function cancelCampaign(uint256 campaignId) external {
        Campaign storage c = campaigns[campaignId];
        require(c.creator != address(0), "Campaign does not exist");
        require(msg.sender == c.creator, "Only creator can cancel");
        require(block.timestamp < c.deadline, "Campaign already ended");
        require(!c.claimed, "Already claimed");
        require(!c.cancelled, "Already cancelled");

        c.cancelled = true;
        emit CampaignCancelled(campaignId);
    }

    // Get full campaign details
    function getCampaign(uint256 campaignId) external view returns (
        address creator,
        string memory title,
        string memory description,
        uint256 goal,
        uint256 raised,
        uint256 deadline,
        bool claimed,
        bool cancelled,
        bool goalReached
    ) {
        Campaign memory c = campaigns[campaignId];
        return (
            c.creator,
            c.title,
            c.description,
            c.goal,
            c.raised,
            c.deadline,
            c.claimed,
            c.cancelled,
            c.raised >= c.goal
        );
    }

    // Get all campaigns by a creator
    function getCreatorCampaigns(address creator) external view returns (uint256[] memory) {
        return creatorCampaigns[creator];
    }

    // Get all campaigns a contributor has funded
    function getContributorCampaigns(address contributor) external view returns (uint256[] memory) {
        return contributorCampaigns[contributor];
    }

    // Check if campaign is still active
    function isActive(uint256 campaignId) external view returns (bool) {
        Campaign memory c = campaigns[campaignId];
        return !c.cancelled && !c.claimed && block.timestamp < c.deadline;
    }

    // Get time remaining in seconds
    function timeRemaining(uint256 campaignId) external view returns (uint256) {
        Campaign memory c = campaigns[campaignId];
        if (block.timestamp >= c.deadline) return 0;
        return c.deadline - block.timestamp;
    }
}
